//  que3::  Find Sum
// You are given an integer N,and your task is to find the sum of all
//  the even integers starting from 1upto N(N inclusive).


const findSum = (n) => {
    
    let sum=0
    let item=2
    while(item<=n){
      sum=sum+item;
      item+=2;
    }
      return sum;
      
      
      
  };
